jenkins-ant-build.xml

发表于 2016-07-01   |  

使用ant打包，我当初也是研究很久才弄出来。我的使用情况：
测试环境和生产环境的配置文件不一样，正式环境就是SVN提交的记录，所以不需要改变，测试环境需要把Test文件夹覆盖resource文件夹

而且ant也需要指定使用buildTest.xml
.
然后我把我的buildTest.xml发出来，仅供参考

<?xml version="1.0" encoding="UTF-8"?>
<project name="lepa-b2c" default="warTest" basedir=".">
    <property environment="env" />
    <property name="src" value="${basedir}/src" />
    <property name="resource" value="${basedir}/resource" />
    <property name="test" value="${basedir}/Test" />
    <property name="bin" value="${basedir}/WebContent/WEB-INF/classes" />
    <property name="build" value="${basedir}/build" />
    <property name="war" value="${build}/war" />
    <property name="lib" value="${basedir}/WebContent/WEB-INF/lib" />
    <property name="captcha" value="${basedir}/src/net/wit/captcha" />

    <path id="javaclass.path">
        <pathelement path="${env.JAVA_HOME}" />
        <fileset dir="${lib}" includes="*.jar" />
    </path>

    <target name="clean">
        <echo>clean...</echo>
        <delete dir="${build}" />
    </target>

    <target name="init">
        <echo>init...</echo>

        <mkdir dir="${war}" />
        <mkdir dir="${bin}" />
    </target>

    <!-- 测试环境编译 -->
    <target name="compileTest" description="Compile Test..." depends="init">
        <javac srcdir="${resource}" destdir="${bin}"  includeantruntime="on"  source="1.7" debug="true" encoding="UTF-8">
            <compilerarg line="-XDignore.symbol.file"/>
            <classpath refid="javaclass.path"></classpath>
        </javac>

        <javac srcdir="${src}" destdir="${bin}"  includeantruntime="on"  source="1.7" debug="true" encoding="UTF-8">
            <compilerarg line="-XDignore.symbol.file"/>
            <classpath refid="javaclass.path"></classpath>
        </javac>

         <copy todir="${bin}" includeemptydirs="false">
            <fileset dir="${resource}">
                <exclude name="**/.svn"/>
            </fileset>
         </copy>

        <copy todir="${bin}" includeemptydirs="false" overwrite="true">
            <fileset dir="${test}">
                <exclude name="**/.svn"/>
            </fileset>
         </copy>

        <copy todir="${bin}/net/wit/captcha" includeemptydirs="false" overwrite="true">
            <fileset dir="${captcha}">
                <exclude name="**/.svn"/>
            </fileset>
         </copy>
    </target>

    <!-- 测试环境打war包 -->
    <target name="warTest" depends="clean,init,compileTest">
        <copy todir="build/war" includeemptydirs="no">
            <fileset dir="WebContent" includes="**/*">
            </fileset>
        </copy>

        <jar destfile="build/ROOT.war">
            <fileset dir="build/war" includes="**/*">
            </fileset>
        </jar>
    </target>

</project>
